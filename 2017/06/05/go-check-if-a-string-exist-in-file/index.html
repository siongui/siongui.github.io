[Golang] Check If a String Exists in File

June 05, 2017
Edit on Github

Check if a string exists in file via Go regexp package.

import (
      "io/ioutil"
      "regexp"
)

func IsExist(str, filepath string) bool {
      b, err := ioutil.ReadFile(filepath)
      if err != nil {
              panic(err)
      }

      isExist, err := regexp.Match(str, b)
      if err != nil {
              panic(err)
      }
      return isExist
}
