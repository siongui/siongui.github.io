Convert the string to 2-dimensional slice.

Solution:

Use strings.Split and strconv.ParseInt in Go standard library to achieve our goal. The following is complete source code.

func parseLine(l string) []int64 {
      var dl []int64

      numbers := strings.Split(l, " ")
      for _, number := range numbers {
              n, err := strconv.ParseInt(number, 10, 0)
              if err != nil {
                      panic(err)
              }
              dl = append(dl, n)
      }
      return dl
}

func parseGrid20x20(g string) {
      lines := strings.Split(g, "\n")
      for _, line := range lines[1 : len(lines)-1] {
              dl := parseLine(line)
              fmt.Println(dl)
      }
}

func main() {
      parseGrid20x20(grid20x20)
}
